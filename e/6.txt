Arduino DHT

pin1 of dht11 with 5v of uno
pin2 of uno with pin 2 of dht11
pin3 of dht11 to gnd

#include <DHT.h>
#define DHTPIN 2
#define DHTTYPE DHT11

DHT dht(DHTPIN, DHTTYPE);

void setup() {
  Serial.begin(9600);
  dht.begin();
}

void loop() {
  float t = dht.readTemperature();
  float h = dht.readHumidity();
  if(isnan(t) || isnan(h)){
    Serial.printlm("Failed to read sensor!");
    return;
  }
  Serial.print("Temp: ");
  Serial.print(t);
  Serial.print(" C, Humidity: ");
  Serial.print(h);
  Serial.println(" %");
  delay(2000);
}
