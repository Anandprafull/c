; bootloader.asm
[BITS 16]
[ORG 0x7C00]

start:
    mov ah, 0x0E
    mov si, loading
.print:
    lodsb
    or al, al
    jz load_kernel
    int 0x10
    jmp .print

load_kernel:
    mov ah, 0x02
    mov al, 1
    mov ch, 0
    mov cl, 2
    mov dh, 0
    mov dl, 0x00
    mov bx, 0x8000
    int 0x13

    jmp 0x0000:0x8000

loading db "Loading kernel...", 0

times 510-($-$$) db 0
dw 0xAA55
